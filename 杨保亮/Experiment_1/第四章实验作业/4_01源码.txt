#include <iostream>

#include "windows.h"
class Student
{
  private:
	  int num;
	  char name[20];
  
  public:
	  static int total;
	  Student() { total++; }
	  ~Student() { total--; }
	  Student(int n, char *p = "wang");
	  void GetName();
	  int Getnum();
  
  };
int Student::total = 0;
Student::Student(int n, char *p)
{
	num = n;
	strcpy(name, p);
	total++;

}
void Student::GetName()
{
	cout << name << endl;
}
int Student::Getnum()
{
	return num;
}
